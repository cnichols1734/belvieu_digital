Embed Code

Add this HTML + JS to the client's website contact page. The form submits in the background and shows a toast notification — no page redirect. 

<form id="bd-contact-form">
  <input type="hidden" name="access_key" value="{{ contact_form_config.access_key }}">

  <label for="name">Name</label>
  <input type="text" name="name" id="name" required>

  <label for="email">Email</label>
  <input type="email" name="email" id="email" required>

  <label for="phone">Phone</label>
  <input type="tel" name="phone" id="phone">

  <label for="message">Message</label>
  <textarea name="message" id="message" required></textarea>

  <button type="submit">Send Message</button>
</form>

<div id="bd-form-toast" style="display:none;position:fixed;top:1.5rem;right:1.5rem;padding:1rem 1.5rem;border-radius:8px;font-size:15px;font-weight:500;z-index:9999;transition:opacity .4s;box-shadow:0 4px 12px rgba(0,0,0,.15);"></div>

<script>
(function() {
  var form = document.getElementById('bd-contact-form');
  var toast = document.getElementById('bd-form-toast');

  function showToast(msg, success) {
    toast.textContent = msg;
    toast.style.background = success ? '#dcfce7' : '#fee2e2';
    toast.style.color = success ? '#166534' : '#991b1b';
    toast.style.border = '1px solid ' + (success ? '#bbf7d0' : '#fecaca');
    toast.style.opacity = '1';
    toast.style.display = 'block';
    setTimeout(function() {
      toast.style.opacity = '0';
      setTimeout(function() { toast.style.display = 'none'; }, 400);
    }, 5000);
  }

  form.addEventListener('submit', function(e) {
    e.preventDefault();
    var btn = form.querySelector('button[type="submit"]');
    var originalText = btn.textContent;
    btn.textContent = 'Sending...';
    btn.disabled = true;

    fetch('{{ app_base_url }}/api/forms/submit', {
      method: 'POST',
      headers: { 'Content-Type': 'application/json' },
      body: JSON.stringify(Object.fromEntries(new FormData(form)))
    })
    .then(function(r) { return r.json(); })
    .then(function(data) {
      if (data.ok) {
        showToast('Message sent! We will be in touch soon.', true);
        form.reset();
      } else {
        showToast(data.error || 'Something went wrong. Please try again.', false);
      }
    })
    .catch(function() {
      showToast('Network error. Please check your connection and try again.', false);
    })
    .finally(function() {
      btn.textContent = originalText;
      btn.disabled = false;
    });
  });
})();
</script>
Style the form and toast however you'd like — only the access_key, field name attributes, and the fetch URL matter.
